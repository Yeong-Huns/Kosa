           decode(n.no, 1, e.job) as job, 
           sum(e.sum_sal) as sum_sal
    from (select deptno, job, sum(sal) as sum_sal
          from emp
          group by deptno, job) e
          CROSS JOIN
         (select level as no
          from dual
          connect by level <= 3) n
    group by decode(n.no, 1, e.deptno, 2, e.deptno),
             decode(n.no, 1, e.job)
order by 1, 2;
