
# Parameter

  https://docs.oracle.com/en/database/oracle/oracle-database/18/lnpls/plsql-subprograms.html

  -> Subprogram Parameters 클릭할 것 
  -> Subprogram Parameter Modes 클릭할 것
  -> Positional, Named, and Mixed Notation for Actual Parameters 

  - (Formal) 매개 변수 : Mode 

    - in (default)
    - out 
    - in out

  - (Actual) 매개 변수 : 값 전달법
     
    - Positional(위치법) Notation 
    - Named(명명법)      Notation 
    - Mixed(조합법)      Notation 


# Actual 매개 변수 예제

  create or replace function addNumberFunc
  (a in number, 
   b in number)
   return number
  is
  begin    
    return a + b;
  end;
  /

  set serveroutput on

  exec dbms_output.put_line(addNumberFunc(100, 200))            : Positional(위치법) Notation 
  exec dbms_output.put_line(addNumberFunc(b => 200, a => 100))  : Named(명명법)      Notation 
  exec dbms_output.put_line(addNumberFunc(100, b => 200))       : Mixed(조합법)      Notation 


# in out 모드 예제

  create or replace procedure sp_in_out_example
  (a in out varchar2)
  is 
  begin
    a := substr(a,1,3)||'-'||substr(a, 4, 4)||'-'||substr(a, 8);
  end;
  / 
  
  -- 테스트
  declare
    telno varchar2(50) := '01023238974';
  begin
    sp_in_out_example(telno);
    dbms_output.put_line(telno);    
  end;
  /
